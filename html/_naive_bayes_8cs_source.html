137  public virtual double CalculateProbabilityOfTokens(IEnumerable<T> Items)
138  {
139  Contract.Requires<ArgumentNullException>(Items != null, "Items");
140  Contract.Requires<InvalidOperationException>(Probabilities != null, "Probabilities has not been initialized");
141  SortedList<string, double> SortedProbabilities = new SortedList<string, double>();
142  int x = 0;
143  foreach (T Item in Items)
144  {
145  double TokenProbability = 0.5;
146  if (Probabilities.ContainsKey(Item))
147  TokenProbability = Probabilities[Item];
148  string Difference = ((0.5 - System.Math.Abs(0.5 - TokenProbability))).ToString(".0000000", CultureInfo.InvariantCulture) + Item + x;
149  SortedProbabilities.Add(Difference, TokenProbability);
150  ++x;
151  }
152  double TotalProbability = 1;
153  double NegativeTotalProbability = 1;
154  int Count = 0;
155  int MaxCount = SortedProbabilities.Count.Min(MaxInterestingTokenCount);
156  foreach (string Probability in SortedProbabilities.Keys)
157  {
158  double TokenProbability = SortedProbabilities[Probability];
159  TotalProbability *= TokenProbability;
160  NegativeTotalProbability *= (1 - TokenProbability);
161  ++Count;
162  if (Count >= MaxCount)
163  break;
164  }
165  return TotalProbability / (TotalProbability + NegativeTotalProbability);
166  }

173  public virtual void LoadTokens(IEnumerable<T> SetATokens, IEnumerable<T> SetBTokens)
174  {
175  Contract.Requires<ArgumentNullException>(SetATokens != null, "SetATokens");
176  Contract.Requires<ArgumentNullException>(SetBTokens != null, "SetBTokens");
177  SetA = SetA.Check(() => new Bag<T>());
178  SetB = SetB.Check(() => new Bag<T>());
179  SetA.Add(SetATokens);
180  SetB.Add(SetBTokens);
181  TotalA = SetA.Sum(x => SetA[x]);
182  TotalB = SetB.Sum(x => SetB[x]);
183  Total = TotalA + TotalB;
184  Probabilities = new ConcurrentDictionary<T, double>();
185  Parallel.ForEach(SetA, Token =>
186  {
187  Probabilities.AddOrUpdate(Token, x => CalculateProbabilityOfToken(x), (x, y) => CalculateProbabilityOfToken(x));
188  });
189  Parallel.ForEach(SetB, Token =>
190  {
191  if (!Probabilities.ContainsKey(Token))
192  Probabilities.AddOrUpdate(Token, x => CalculateProbabilityOfToken(x), (x, y) => y);
193  });
194  }

201  protected virtual double CalculateProbabilityOfToken(T Item)
202  {
203  Contract.Requires<ArgumentNullException>(SetA != null && SetB != null, "Probabilities have not been initialized");
204  double Probability = 0;
205  int ACount = SetA.Contains(Item) ? SetA[Item] * ATokenWeight : 0;
206  int BCount = SetB.Contains(Item) ? SetB[Item] * BTokenWeight : 0;
207  if (ACount + BCount >= MinCountForInclusion)
208  {
209  double AProbability = ((double)ACount / (double)TotalA).Min(1);
210  double BProbability = ((double)BCount / (double)TotalB).Min(1);
211  Probability = MinTokenProbability.Max(MaxTokenProbability.Min(AProbability / (AProbability + BProbability)));
212  }
213  return Probability;
214  }
