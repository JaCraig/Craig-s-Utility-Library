150  public IList<IList<dynamic>> Execute()
151  {
152  IList<IList<dynamic>> ReturnValue = new List<IList<dynamic>>();
153  if (Commands.Count == 0)
154  {
155  ReturnValue.Add(new List<dynamic>());
156  return ReturnValue;
157  }
158  using (DirectoryEntry Entry = new DirectoryEntry(Source.Server, Source.UserName, Source.Password, AuthenticationTypes.Secure))
159  {
160  using (DirectorySearcher Searcher = new DirectorySearcher(Entry))
161  {
162  Searcher.PageSize = 1000;
163  foreach (Command Command in Commands)
164  {
165  Searcher.Filter = Command.SQLCommand;
166  using (SearchResultCollection Results = Searcher.FindAll())
167  {
168  List<dynamic> ReturnValues = new List<dynamic>();
169  foreach (SearchResult Result in Results)
170  {
171  dynamic TempValue = new Dynamo();
172  foreach (PropertyValueCollection Property in Result.GetDirectoryEntry().Properties)
173  {
174  TempValue[Property.PropertyName] = Property.Value;
175  ReturnValues.Add(TempValue);
176  }
177  }
178  ReturnValue.Add(ReturnValues);
179  }
180  }
181  }
182  Entry.Close();
183  }
184  for (int x = 0; x < Commands.Count(); ++x)
185  {
186  Commands[x].Finalize(ReturnValue[x]);
187  }
188  return ReturnValue;
189  }
