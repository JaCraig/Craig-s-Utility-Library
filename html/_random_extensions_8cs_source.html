143  public static T Next<T>(this System.Random Random, IEnumerable<T> List)
144  {
145  Contract.Requires<ArgumentNullException>(Random != null, "Random");
146  if (List == null)
147  return default(T);
148  int x = 0;
149  int Position = Random.Next(0, List.Count());
150  foreach (T Item in List)

152  if (x == Position)
153  return Item;
154  ++x;
