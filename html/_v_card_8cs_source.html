302  protected override void LoadFromData(string Data)
303  {
304  string Content = Data;
305  foreach (Match TempMatch in Regex.Matches(Content, "(?<Title>[^:]+):(?<Value>.*)"))
306  {
307  switch (TempMatch.Groups["Title"].Value.ToUpperInvariant())
308  {
309  case "N":
310  string[] Name = TempMatch.Groups["Value"].Value.Split(';');
311  if (Name.Length > 0)
312  {
313  LastName = Name[0];
314  if (Name.Length > 1)
315  FirstName = Name[1];
316  if (Name.Length > 2)
317  MiddleName = Name[2];
318  if (Name.Length > 3)
319  Prefix = Name[3];
320  if (Name.Length > 4)
321  Suffix = Name[4];
322  }
323 
324  break;
325  case "TEL;WORK":
326  DirectDial = TempMatch.Groups["Value"].Value;
327  break;
328  case "EMAIL;INTERNET":
329  Email = TempMatch.Groups["Value"].Value;
330  break;
331  case "TITLE":
332  Title = TempMatch.Groups["Value"].Value;
333  break;
334  case "ORG":
335  Organization = TempMatch.Groups["Value"].Value;
336  break;
337  }
338  }
