106  public void Parse(string FileContent)
107  {
108  if (string.IsNullOrEmpty(FileContent))
109  return;
110  Regex TempSplitter = new Regex("[^\"\r\n]*(\r\n|\n|$)|(([^\"\r\n]*)(\"[^\"]*\")([^\"\r\n]*))*(\r\n|\n|$)");
111  MatchCollection Matches = TempSplitter.Matches(FileContent);
112  if (string.IsNullOrEmpty(Delimiter) && Matches != null)
113  Delimiter = CheckDelimiters(Matches.Where(x => !string.IsNullOrEmpty(x.Value)).FirstOrDefault().Chain(x => x.Value, ","));
114  Matches.Where(x => !string.IsNullOrEmpty(x.Value))
115  .ForEach(x => Records.Add(new Row(x.Value, Delimiter)));
116  }

173  private static string CheckDelimiters(string Content)
174  {
175  if (string.IsNullOrEmpty(Content))
176  return ",";
177  string[] Delimiters = new string[] { ",", "|", "\t", "$", ";", ":" };
178  int[] Count = new int[6];
179  int MaxIndex = 0;
180  for (int x = 0; x < Delimiters.Length; ++x)
181  {
182  string TempDelimiter = Delimiters[x];
183  Regex TempSplitter = new Regex(string.Format(CultureInfo.InvariantCulture, "(?<Value>\"(?:[^\"]|\"\")*\"|[^{0}\r\n]*?)(?<Delimiter>{0}|\r\n|\n|$)", Regex.Escape(TempDelimiter)));
184  Count[x] = TempSplitter.Matches(Content).Count;
185  if (Count[MaxIndex] < Count[x])
186  MaxIndex = x;
187  }
188  return Count[MaxIndex] > 1 ? Delimiters[MaxIndex] : ",";
189  }
