312  public static bool Is(this DateTime Date, DateCompare Comparison)
313  {
314  if (Comparison.HasFlag(DateCompare.InFuture) && DateTime.Now >= Date)
315  return false;
316  if (Comparison.HasFlag(DateCompare.InPast) && DateTime.Now <= Date)
317  return false;
318  if (Comparison.HasFlag(DateCompare.Today) && DateTime.Today != Date.Date)
319  return false;
320  if (Comparison.HasFlag(DateCompare.WeekDay) && ((int)Date.DayOfWeek == 6 || (int)Date.DayOfWeek == 0))
321  return false;
322  if (Comparison.HasFlag(DateCompare.WeekEnd) && (int)Date.DayOfWeek != 6 && (int)Date.DayOfWeek != 0)
323  return false;
324  return true;
325  }
