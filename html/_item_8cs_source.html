50  public Item(IXPathNavigable Doc)
51  : this()
52  {
53  Contract.Requires<ArgumentNullException>(Doc != null, "Doc");
54  XPathNavigator Element = Doc.CreateNavigator();
55  var NamespaceManager = new XmlNamespaceManager(Element.NameTable);
56  NamespaceManager.AddNamespace("media", "http://search.yahoo.com/mrss/");
57  XPathNavigator Node = Element.SelectSingleNode("./title", NamespaceManager);
58  if (Node != null)
59  {
60  Title = Node.Value;
61  }
62  Node = Element.SelectSingleNode("./link", NamespaceManager);
63  if (Node != null)
64  {
65  Link = Node.Value;
66  }
67  Node = Element.SelectSingleNode("./description", NamespaceManager);
68  if (Node != null)
69  {
70  Description = Node.Value;
71  }
72  Node = Element.SelectSingleNode("./author", NamespaceManager);
73  if (Node != null)
74  {
75  Author = Node.Value;
76  }
77  XPathNodeIterator Nodes = Element.Select("./category", NamespaceManager);
78  foreach (XmlNode TempNode in Nodes)
79  {
80  Categories.Add(Utils.StripIllegalCharacters(TempNode.Value));
81  }
82  Node = Element.SelectSingleNode("./enclosure", NamespaceManager);
83  if (Node != null)

86  }
87  Node = Element.SelectSingleNode("./pubdate", NamespaceManager);

169  if (GUID != null)
