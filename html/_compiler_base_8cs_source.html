147  protected IEnumerable<Type> Add(string Code, IEnumerable<string> Usings, params Assembly[] References)
148  {
149  if (AssemblyStream == null)
150  return null;
151  CSharpCompilation CSharpCompiler = CSharpCompilation.Create(AssemblyName + ".dll",
152  new SyntaxTree[] { CSharpSyntaxTree.ParseText(Code) },
153  References.ForEach(x => MetadataReference.CreateFromFile(x.Location)),
154  new CSharpCompilationOptions(OutputKind.DynamicallyLinkedLibrary, usings: Usings, optimizationLevel: Optimize ? OptimizationLevel.Release : OptimizationLevel.Debug));
155  using (MemoryStream TempStream = new MemoryStream())
156  {
157  EmitResult Result = CSharpCompiler.Emit(TempStream);
158  if (!Result.Success)
159  throw new Exception(Code + System.Environment.NewLine + System.Environment.NewLine + Result.Diagnostics.ToString(x => x.GetMessage() + " : " + x.Location.GetLineSpan().StartLinePosition.Line, System.Environment.NewLine));
160  byte[] MiniAssembly = TempStream.ToArray();
161  Classes.AddIfUnique((x, y) => x.FullName == y.FullName, AppDomain.CurrentDomain.Load(MiniAssembly).GetTypes());
162  AssemblyStream.Write(MiniAssembly, 0, MiniAssembly.Length);
163  }
