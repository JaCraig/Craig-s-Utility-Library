54  public Profiler(string FunctionName)
55  {
56  this.Parent = Current;
57  Profiler Child = Parent != null && Parent.InternalChildren.ContainsKey(FunctionName) ? Parent.InternalChildren[FunctionName] : null;
58  if (Child == null)
59  {
60  if (Parent != null)
61  Parent.InternalChildren.Add(FunctionName, this);
62  this.Function = FunctionName;
63  this.InternalChildren = new Dictionary<string, Profiler>();
64  this.Entries = new List<IResultEntry>();
65  this.StopWatch = new StopWatch();
66  this.Level = Parent == null ? 0 : Parent.Level + 1;
67  this.CalledFrom = new StackTrace().GetMethods(this.GetType().Assembly).ToString<MethodBase>(x => x.DeclaringType.Name + " > " + x.Name, "<br />");
68  Running = false;
69  Current = this;
70  Child = this;
71  if (CPUCounter == null)
72  CPUCounter = new PerformanceCounter("Processor Information", "% Processor Time", "_Total");
73  if (CounterStopWatch == null)
74  {
75  CounterStopWatch = new StopWatch();
76  CounterStopWatch.Start();
77  LastCounterTime = CounterStopWatch.ElapsedTime;
78  }
79  }
80  else
81  {
82  Current = Child;
83  }

85  }
