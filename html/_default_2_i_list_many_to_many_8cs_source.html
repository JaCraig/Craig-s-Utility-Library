82  public override IBatch CascadeDelete(ClassType Object, ISourceInfo Source, IList<object> ObjectsSeen)
83  {
84  QueryProvider.Manager Provider = IoC.Manager.Bootstrapper.Resolve<QueryProvider.Manager>();
85  Mapper.Manager MappingProvider = IoC.Manager.Bootstrapper.Resolve<Mapper.Manager>();
86  IMapping PropertyMapping = MappingProvider[typeof(DataType), Source];
87  IBatch Batch = Provider.Batch(Source);
88  var AspectObject = Object as IORMObject;
89  if (Object == null || (AspectObject != null && ObjectsSeen.Contains(Mapping.IDProperties.FirstOrDefault().GetValue(Object))))
90  return Batch;
91  if (AspectObject != null)
92  ObjectsSeen.Add(Mapping.IDProperties.FirstOrDefault().GetValue(Object));
93  IEnumerable<DataType> IList = CompiledExpression(Object);
94  if (IList == null)
95  return Batch;
96  foreach (DataType Item in IList.Where(x => x != null))
97  {
98  foreach (IProperty<DataType> Property in PropertyMapping.Properties.Where(x => x.Cascade))
99  {
100  Batch.AddCommand(Property.CascadeDelete(Item, Source, ObjectsSeen.ToList()));
101  }
102  Batch.AddCommand(Provider.Generate<DataType>(Source, PropertyMapping).Delete(Item));
103  }
104  Utilities.IoC.Manager.Bootstrapper.Resolve<DataTypes.Caching.Manager>().Cache().RemoveByTag(typeof(DataType).GetName());
105  return Batch;
106  }
