114  private static object SqlDbTypeToType(object arg)
115  {
116  if (!(arg is SqlDbType))
117  return typeof(int);
118  SqlDbType Item = (SqlDbType)arg;
119  SqlParameter Parameter = new SqlParameter();
120  Parameter.SqlDbType = Item;
121  if (Parameter.DbType == DbType.Byte) return typeof(byte);
122  else if (Parameter.DbType == DbType.SByte) return typeof(sbyte);
123  else if (Parameter.DbType == DbType.Int16) return typeof(short);
124  else if (Parameter.DbType == DbType.UInt16) return typeof(ushort);
125  else if (Parameter.DbType == DbType.Int32) return typeof(int);
126  else if (Parameter.DbType == DbType.UInt32) return typeof(uint);
127  else if (Parameter.DbType == DbType.Int64) return typeof(long);
128  else if (Parameter.DbType == DbType.UInt64) return typeof(ulong);
129  else if (Parameter.DbType == DbType.Single) return typeof(float);
130  else if (Parameter.DbType == DbType.Double) return typeof(double);
131  else if (Parameter.DbType == DbType.Decimal) return typeof(decimal);
132  else if (Parameter.DbType == DbType.Boolean) return typeof(bool);
133  else if (Parameter.DbType == DbType.String) return typeof(string);
134  else if (Parameter.DbType == DbType.StringFixedLength) return typeof(char);
135  else if (Parameter.DbType == DbType.Guid) return typeof(Guid);
136  else if (Parameter.DbType == DbType.DateTime2) return typeof(DateTime);
137  else if (Parameter.DbType == DbType.DateTime) return typeof(DateTime);
138  else if (Parameter.DbType == DbType.DateTimeOffset) return typeof(DateTimeOffset);
139  else if (Parameter.DbType == DbType.Binary) return typeof(byte[]);
140  return typeof(int);
