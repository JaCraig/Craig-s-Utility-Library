237  private void Load(IXPathNavigable Document)
238  {
239  Contract.Requires<ArgumentNullException>(Document != null, "Document");
240  Channels = Channels.Check(new List<Channel>());
241  XPathNavigator Navigator = Document.CreateNavigator();
242  XmlNamespaceManager NamespaceManager = new XmlNamespaceManager(Navigator.NameTable);
243  XPathNodeIterator Nodes = Navigator.Select("./channel", NamespaceManager);
244  foreach (XmlNode Element in Nodes)
245  {
246  Channels.Add(new Channel((XmlElement)Element));
247  }
248  if (Channels.Count == 0)
249  {
250  Nodes = Navigator.Select(".//channel", NamespaceManager);
251  foreach (XmlNode Element in Nodes)
252  {
253  Channels.Add(new Channel((XmlElement)Element));
254  }
255  List<Item> Items = new List<Item>();
256  Nodes = Navigator.Select(".//item", NamespaceManager);
257  foreach (XmlNode Element in Nodes)
258  {
259  Items.Add(new Item((XmlElement)Element));
260  }
261  if (Channels.Count > 0)
262  {
263  Channels.FirstOrDefault().Add(Items);
264  }
265  }
266  }
