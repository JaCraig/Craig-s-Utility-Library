277  private ConstructorInfo FindConstructor(Type Type)
278  {
279  Contract.Requires<ArgumentNullException>(Type != null, "Type");
280  ConstructorInfo[] Constructors = Type.GetConstructors();
281  ConstructorInfo Constructor = null;
282  foreach (ConstructorInfo TempConstructor in Constructors.OrderByDescending(x => x.GetParameters().Length))
283  {
284  bool Found = true;
285  foreach (ParameterInfo Parameter in TempConstructor.GetParameters())
286  {
287  Type ParameterType = Parameter.ParameterType;
288  if (Parameter.ParameterType.GetInterfaces().Contains(typeof(IEnumerable)) && Parameter.ParameterType.IsGenericType)
289  {
290  ParameterType = ParameterType.GetGenericArguments().First();
291  if (!AppContainer.Keys.Any(x => x.Item1 == ParameterType))
292  {
293  Found = false;
294  break;
295  }
296  }
297  else if (!this.AppContainer.Keys.Contains(new Tuple<Type, string>(ParameterType, "")))
298  {
299  Found = false;
300  break;
301  }
302  }
303  if (Found)
304  {
305  Constructor = TempConstructor;
306  break;
307  }
308  }
309  return Constructor;
310  }

317  private List<object> GetParameters(ConstructorInfo Constructor)
318  {
319  Contract.Requires<ArgumentNullException>(Constructor != null, "Constructor");
320  List<object> Params = new List<object>();
321  foreach (ParameterInfo Parameter in Constructor.GetParameters())
322  {
323  if (Parameter.ParameterType.GetInterfaces().Contains(typeof(IEnumerable)) && Parameter.ParameterType.IsGenericType)
324  {
325  Type GenericParamType = Parameter.ParameterType.GetGenericArguments().First();
326  Params.Add(GenericResolveAllMethod.MakeGenericMethod(GenericParamType).Invoke(this, new object[] { }));
327  }
328  else
329  {

333  }));
334  }
335  }
