70  public override IBatch CascadeDelete(ClassType Object, ISourceInfo Source, IList<object> ObjectsSeen)
71  {
72  QueryProvider.Manager Provider = IoC.Manager.Bootstrapper.Resolve<QueryProvider.Manager>();
73  Mapper.Manager MappingProvider = IoC.Manager.Bootstrapper.Resolve<Mapper.Manager>();
74  IMapping PropertyMapping = MappingProvider[typeof(DataType), Source];
75  IBatch Batch = Provider.Batch(Source);
76  var AspectObject = Object as IORMObject;
77  if (Object == null || (AspectObject != null && ObjectsSeen.Contains(Mapping.IDProperties.FirstOrDefault().GetValue(Object))))
78  return Provider.Batch(Source);
79  if (AspectObject != null)
80  ObjectsSeen.Add(Mapping.IDProperties.FirstOrDefault().GetValue(Object));
81  DataType Item = CompiledExpression(Object);
82  if (Item == null)
83  return Batch;
84  foreach (IProperty<DataType> Property in PropertyMapping.Properties.Where(x => x.Cascade))
85  {
86  Batch.AddCommand(Property.CascadeDelete(Item, Source, ObjectsSeen.ToList()));
87  }
88  Batch.AddCommand(Provider.Generate<DataType>(Source, PropertyMapping).Delete(Item));
89  Utilities.IoC.Manager.Bootstrapper.Resolve<DataTypes.Caching.Manager>().Cache().RemoveByTag(typeof(DataType).GetName());
90  return Batch;
91  }
