104  public virtual string GetHCalendar()
105  {
106  var Output = new StringBuilder();
107  Output.Append("<div class=\"vevent\">")
108  .Append("<div class=\"summary\">").Append(Subject).Append("</div>")
109  .Append("<div>Date: <abbr class=\"dtstart\" title=\"")
110  .Append(StartTime.ToString("MM-dd-yyyy hh:mm tt", CultureInfo.InvariantCulture)).Append("\">")
111  .Append(StartTime.ToString("MMMM dd, yyyy hh:mm tt", CultureInfo.InvariantCulture)).Append("</abbr> to ")
112  .Append("<abbr class=\"dtend\" title=\"").Append(EndTime.ToString("MM-dd-yyyy hh:mm tt", CultureInfo.InvariantCulture))
113  .Append("\">");
114  if (EndTime.Year != StartTime.Year)
115  {
116  Output.Append(EndTime.ToString("MMMM dd, yyyy hh:mm tt", CultureInfo.CurrentCulture));
117  }
118  else if (EndTime.Month != StartTime.Month)
119  {
120  Output.Append(EndTime.ToString("MMMM dd hh:mm tt", CultureInfo.CurrentCulture));
121  }
122  else if (EndTime.Day != StartTime.Day)
123  {
124  Output.Append(EndTime.ToString("dd hh:mm tt", CultureInfo.CurrentCulture));
125  }
126  else
127  {
128  Output.Append(EndTime.ToString("hh:mm tt", CultureInfo.CurrentCulture));
129  }
130  return Output.Append("</abbr></div>")
131  .Append("<div>Location: <span class=\"location\">").Append(Location).Append("</span></div>")
132  .Append("<div class=\"description\">").Append(Description).Append("</div>")
133  .Append("</div>")
134  .ToString();
135  }

246  protected override void LoadFromData(string Data)
247  {
248  foreach (Match TempMatch in Regex.Matches(Data, "(?<Title>[^\r\n:]+):(?<Value>[^\r\n]*)"))
249  {
250  if (TempMatch.Groups["Title"].Value.ToUpperInvariant() == "DTSTART")
251  {
252  StartTime = CurrentTimeZone.ToLocalTime(DateTime.Parse(TempMatch.Groups["Value"].Value.ToString(@"####/##/## ##:##"), CultureInfo.CurrentCulture));
253  }
254  else if (TempMatch.Groups["Title"].Value.ToUpperInvariant() == "DTEND")
255  {
256  EndTime = CurrentTimeZone.ToLocalTime(DateTime.Parse(TempMatch.Groups["Value"].Value.ToString(@"####/##/## ##:##"), CultureInfo.CurrentCulture));
257  }
258  else if (TempMatch.Groups["Title"].Value.ToUpperInvariant() == "LOCATION")
259  {
260  this.Location = TempMatch.Groups["Value"].Value;
261  }
262  else if (TempMatch.Groups["Title"].Value.ToUpperInvariant() == "SUMMARY;LANGUAGE=EN-US")
263  {
264  Subject = TempMatch.Groups["Value"].Value;
265  }
266  else if (TempMatch.Groups["Title"].Value.ToUpperInvariant() == "DESCRIPTION" && string.IsNullOrEmpty(Description))
267  {
268  Description = TempMatch.Groups["Value"].Value;
269  }
270  }
