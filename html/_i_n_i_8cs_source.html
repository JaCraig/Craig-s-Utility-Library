184  protected override void LoadFromData(string Data)
185  {
186  FileContents = new Dictionary<string, Dictionary<string, string>>();
187  string Contents = Data;
188  Regex Section = new Regex("[" + Regex.Escape(" ") + "\t]*" + Regex.Escape("[") + ".*" + Regex.Escape("]\r\n"));
189  string[] Sections = Section.Split(Contents);
190  MatchCollection SectionHeaders = Section.Matches(Contents);
191  int Counter = 1;
192  foreach (Match SectionHeader in SectionHeaders)
193  {
194  string[] Splitter = { "\r\n" };
195  string[] Splitter2 = { "=" };
196  string[] Items = Sections[Counter].Split(Splitter, StringSplitOptions.RemoveEmptyEntries);
197  Dictionary<string, string> SectionValues = new Dictionary<string, string>();
198  foreach (string Item in Items)
199  {
200  string[] ItemSplit = Item.Split(Splitter2, StringSplitOptions.None);
201  if (ItemSplit.Length >= 2)
202  SectionValues.Add(ItemSplit[0], ItemSplit[1]);
203  }
204  FileContents.Add(SectionHeader.Value.Replace("[", "").Replace("]\r\n", ""), SectionValues);
205  ++Counter;
206  }
207  }
