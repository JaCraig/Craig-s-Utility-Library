225  private static string SendRequest(HttpWebRequest Request)
226  {
227  Contract.Requires<ArgumentNullException>(Request != null, "Request");
228  using (HttpWebResponse Response = Request.GetResponse() as HttpWebResponse)
229  {
230  if (Response.StatusCode != HttpStatusCode.OK)
231  return "";
232  using (StreamReader Reader = new StreamReader(Response.GetResponseStream()))
233  {
234  return Reader.ReadToEnd();
235  }
236  }

244  private static void SetupData(HttpWebRequest Request, string Data)
245  {
246  Contract.Requires<ArgumentNullException>(Request != null, "Request");
247  if (string.IsNullOrEmpty(Data))
