106  public static List<T> To<T>(this DataTable Data, Func<T> Creator)
107  where T : class,new()
108  {
109  if (Data == null)
110  return new List<T>();
111  Creator = Creator.Check(() => new T());
112  Type TType = typeof(T);
113  PropertyInfo[] Properties = TType.GetProperties();
114  var Results = new List<T>();
115  for (int x = 0; x < Data.Rows.Count; ++x)
116  {
117  T RowObject = Creator();
118  for (int y = 0; y < Data.Columns.Count; ++y)
119  {
120  PropertyInfo Property = Properties.FirstOrDefault(z => z.Name == Data.Columns[y].ColumnName);
121  if (Property != null)
122  Property.SetValue(RowObject, Data.Rows[x][Data.Columns[y]].To(Property.PropertyType, null), new object[] { });
123  }
124  Results.Add(RowObject);
125  }
126  return Results;
127  }
