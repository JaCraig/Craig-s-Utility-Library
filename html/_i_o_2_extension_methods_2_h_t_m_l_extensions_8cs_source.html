114  private static string CSSMinify(string Input)
115  {
116  if (string.IsNullOrEmpty(Input))
117  return "";
118  Input = Regex.Replace(Input, @"(/\*\*/)|(/\*[^!][\s\S]*?\*/)", string.Empty);
119  Input = Regex.Replace(Input, @"\s+", " ");
120  Input = Regex.Replace(Input, @"(\s([\{:,;\}\(\)]))", "$2");
121  Input = Regex.Replace(Input, @"(([\{:,;\}\(\)])\s)", "$2");
122  Input = Regex.Replace(Input, ":0 0 0 0;", ":0;");
123  Input = Regex.Replace(Input, ":0 0 0;", ":0;");
124  Input = Regex.Replace(Input, ":0 0;", ":0;");
125  Input = Regex.Replace(Input, ";}", "}");
126  Input = Regex.Replace(Input, @"(?<=[>])\s{2,}(?=[<])|(?<=[>])\s{2,}(?=&nbsp;)|(?<=&nbsp;)\s{2,}(?=[<])", string.Empty);
127  Input = Regex.Replace(Input, @"([!{}:;>+([,])\s+", "$1");
128  Input = Regex.Replace(Input, @"([\s:])(0)(px|em|%|in|cm|mm|pc|pt|ex)", "$1$2");
129  Input = Regex.Replace(Input, "background-position:0", "background-position:0 0");
130  Input = Regex.Replace(Input, @"(:|\s)0+\.(\d+)", "$1.$2");
131  Input = Regex.Replace(Input, @"[^\}]+\{;\}", "");
132  return Input;
133  }

145  private static string HTMLMinify(string Input)
146  {
147  if (string.IsNullOrEmpty(Input))
148  return "";
149  Input = Regex.Replace(Input, "/// <.+>", "");
150  if (string.IsNullOrEmpty(Input))
151  return "";
152  Input = Regex.Replace(Input, @">[\s\S]*?<", new MatchEvaluator(Evaluate));
153  return Input;
154  }

156  private static string JavaScriptMinify(string Input)
157  {
158  if (string.IsNullOrEmpty(Input))
159  return "";
160  string[] CodeLines = Input.Split(new string[] { System.Environment.NewLine, "\n" }, StringSplitOptions.RemoveEmptyEntries);
161  StringBuilder Builder = new StringBuilder();
162  foreach (string Line in CodeLines)
163  {
164  string Temp = Line.Trim();
165  if (Temp.Length > 0 && !Temp.StartsWith("//", StringComparison.InvariantCulture))
166  Builder.AppendLine(Temp);
167  }
168 
169  Input = Builder.ToString();
170  Input = Regex.Replace(Input, @"(/\*\*/)|(/\*[^!][\s\S]*?\*/)", string.Empty);
171  Input = Regex.Replace(Input, @"^[\s]+|[ \f\r\t\v]+$", String.Empty);
172  Input = Regex.Replace(Input, @"^[\s]+|[ \f\r\t\v]+$", String.Empty);
173  Input = Regex.Replace(Input, @"([+-])\n\1", "$1 $1");
174  Input = Regex.Replace(Input, @"([^+-][+-])\n", "$1");
175  Input = Regex.Replace(Input, @"([^+]) ?(\+)", "$1$2");
176  Input = Regex.Replace(Input, @"(\+) ?([^+])", "$1$2");
177  Input = Regex.Replace(Input, @"([^-]) ?(\-)", "$1$2");
178  Input = Regex.Replace(Input, @"(\-) ?([^-])", "$1$2");
179  Input = Regex.Replace(Input, @"\n([{}()[\],<>/*%&|^!~?:=.;+-])", "$1");
180  Input = Regex.Replace(Input, @"(\W(if|while|for)\([^{]*?\))\n", "$1");
181  Input = Regex.Replace(Input, @"(\W(if|while|for)\([^{]*?\))((if|while|for)\([^{]*?\))\n", "$1$3");
182  Input = Regex.Replace(Input, @"([;}]else)\n", "$1 ");
183  Input = Regex.Replace(Input, @"(?<=[>])\s{2,}(?=[<])|(?<=[>])\s{2,}(?=&nbsp;)|(?<=&nbsp;)\s{2,}(?=[<])", String.Empty);
184 
185  return Input;
186  }
