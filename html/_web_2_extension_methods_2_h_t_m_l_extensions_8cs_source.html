115  public static void HTTPCompress(this HttpContextBase Context, bool RemovePrettyPrinting = false, MinificationType Type = MinificationType.HTML)
116  {
117  Contract.Requires<ArgumentNullException>(Context != null, "Context");
118  if (Context.Request.UserAgent != null && Context.Request.UserAgent.Contains("MSIE 6"))
119  return;
120  if (RemovePrettyPrinting)
121  {
122  if (Context.IsEncodingAccepted(GZIP))
123  {
124  Context.Response.Filter = new UglyStream(Context.Response.Filter, CompressionType.GZip, Type);

337  private static string RemoveDiacritics(string Input)
338  {
339  if (string.IsNullOrEmpty(Input))
340  return Input;
341  string Normalized = Input.Normalize(NormalizationForm.FormD);
342  StringBuilder Builder = new StringBuilder();
343  for (int i = 0; i < Normalized.Length; i++)
344  {
345  Char TempChar = Normalized[i];
346  if (CharUnicodeInfo.GetUnicodeCategory(TempChar) != UnicodeCategory.NonSpacingMark)
347  Builder.Append(TempChar);
348  }
349  return Builder.ToString();
350  }
