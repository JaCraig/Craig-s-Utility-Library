180  private IList<IList<dynamic>> ExecuteCommands()
181  {
182  Contract.Requires(this.Source != null);
183  if (Commands == null)
184  return new List<IList<dynamic>>();
185  IList<IList<dynamic>> ReturnValue = new List<IList<dynamic>>();
186  if (Commands.Count == 0)
187  {
188  ReturnValue.Add(new List<dynamic>());
189  return ReturnValue;
190  }
191  DbProviderFactory Factory = DbProviderFactories.GetFactory(Source.SourceType);
192  using (DbConnection Connection = Factory.CreateConnection())
193  {
194  Connection.ConnectionString = Source.Connection;
195  using (DbCommand ExecutableCommand = Factory.CreateCommand())
196  {
197  ExecutableCommand.Connection = Connection;
198  ExecutableCommand.CommandType = CommandType.Text;
199  if (Commands.Count > 1
200  && !Commands.Any(x => x.SQLCommand.Contains("ALTER DATABASE"))
201  && !Commands.Any(x => x.SQLCommand.Contains("CREATE DATABASE")))
202  ExecutableCommand.BeginTransaction();
203  ExecutableCommand.Open();
204 
205  try
206  {
207  int Count = 0;
208  while (true)
209  {
210  List<IParameter> FinalParameters = new List<IParameter>();
211  string FinalSQLCommand = "";
212  int ParameterTotal = 0;
213  ExecutableCommand.Parameters.Clear();
214  for (int y = Count; y < Commands.Count; ++y)
215  {
216  ICommand Command = Commands[y];
217  if (ParameterTotal + Command.Parameters.Count > 2100)
218  break;
219  ParameterTotal += Command.Parameters.Count;
220  if (Command.CommandType == System.Data.CommandType.Text)
221  {
222  FinalSQLCommand += string.IsNullOrEmpty(Command.SQLCommand) ?
223  "" :
224  ParameterRegex.Replace(Command.SQLCommand, x =>
225  {
226  if (!x.Value.StartsWith("@@", StringComparison.Ordinal))
227  return x.Value + "Command" + Count.ToString(CultureInfo.InvariantCulture);
228  return x.Value;
229  }) + System.Environment.NewLine;
230  foreach (IParameter Parameter in Command.Parameters)
231  {
232  FinalParameters.Add(Parameter.CreateCopy("Command" + Count.ToString(CultureInfo.InvariantCulture)));
233  }
234  }
235  else
236  {
237  FinalSQLCommand += Command.SQLCommand + System.Environment.NewLine;
238  foreach (IParameter Parameter in Command.Parameters)
239  {
240  FinalParameters.Add(Parameter.CreateCopy(""));
241  }
242  }
243  ++Count;
244  }
245 
246  ExecutableCommand.CommandText = FinalSQLCommand;
247  FinalParameters.ForEach(x => x.AddParameter(ExecutableCommand));
248 
249  using (DbDataReader TempReader = ExecutableCommand.ExecuteReader())
250  {
251  ReturnValue.Add(GetValues(TempReader));
252  while (TempReader.NextResult())
253  {
254  ReturnValue.Add(GetValues(TempReader));
255  }
256  }
257  if (Count >= CommandCount)
258  break;
259  }
260  ExecutableCommand.Commit();
261  }
262  catch { ExecutableCommand.Rollback(); throw; }
263  finally { ExecutableCommand.Close(); }
264  }
265  }
266  for (int x = 0, y = 0; x < Commands.Count(); ++x)
267  {
268  if (Commands[x].Finalizable)
269  {
270  Commands[x].Finalize(ReturnValue[y]);
271  ++y;
272  }
273  else
274  Commands[x].Finalize(new List<dynamic>());
275  }
276  return ReturnValue;
277  }
