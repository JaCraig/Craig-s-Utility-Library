92  public ISource GetSourceStructure(ISourceInfo Source)
93  {
94  var DatabaseName = Regex.Match(Source.Connection, "Initial Catalog=(.*?;)").Value.Replace("Initial Catalog=", "").Replace(";", "");
95  var DatabaseSource = SourceProvider.GetSource(Regex.Replace(Source.Connection, "Initial Catalog=(.*?;)", ""));
96  if (!SourceExists(DatabaseName, DatabaseSource))
97  return null;
98  var Temp = new Database(DatabaseName);
99  var Batch = Provider.Batch(Source);
100  IBuilder[] Builders = {
101  new Tables(),
102  new TableColumns(),
103  new TableTriggers(),
104  new TableForeignKeys(),
105  new Views(),
106  new StoredProcedures(),
107  new StoredProcedureColumns(),
108  new Functions()
109  };
110  Builders.ForEach(x => x.GetCommand(Batch));
111  var Results = Batch.Execute();
112  Builders.For(0, Builders.Length - 1, (x, y) => y.FillDatabase(Results[x], Temp));
113  return Temp;
114  }

123  public void Setup(ListMapping<IDatabase, IMapping> Mappings, IDatabase Database, QueryProvider.Manager QueryProvider, Graph<IMapping> Structure)
124  {

127  SetupTables(Mappings, Database, TempDatabase, Structure);
128  SetupJoiningTables(Mappings, Database, TempDatabase);

132  {
133  Table.SetupForeignKeys();
134  }
135  var Commands = GenerateSchema(TempDatabase, SourceProvider.GetSource(Database.Name)).ToList();
136  var Batch = QueryProvider.Batch(SourceProvider.GetSource(Database.Name));
137  for (int x = 0; x < Commands.Count; ++x)
138  {
139  if (Commands[x].ToUpperInvariant().Contains("CREATE DATABASE"))
140  {
141  QueryProvider.Batch(SourceProvider.GetSource(Regex.Replace(SourceProvider.GetSource(Database.Name).Connection, "Initial Catalog=(.*?;)", ""))).AddCommand(null, null, CommandType.Text, Commands[x]).Execute();
142  }

144  {
145  if (Batch.CommandCount > 0)
146  {
147  Batch.Execute();
148  Batch = QueryProvider.Batch(SourceProvider.GetSource(Database.Name));
149  }
150  Batch.AddCommand(null, null, CommandType.Text, Commands[x]);
151  if (x < Commands.Count - 1)
152  {
153  Batch.Execute();

155  }
156  }

220  private static IEnumerable<string> BuildCommands(ISource DesiredStructure, ISource CurrentStructure)
221  {
222  var Commands = new List<string>();
223  DesiredStructure = DesiredStructure.Check(new Database(""));
224  if (CurrentStructure == null)
225  Commands.Add(string.Format(CultureInfo.CurrentCulture,
226  "EXEC dbo.sp_executesql @statement = N'CREATE DATABASE {0}'",
227  DesiredStructure.Name));
228  CurrentStructure = CurrentStructure.Check(new Database(DesiredStructure.Name));
229  foreach (Table Table in DesiredStructure.Tables)
230  {
231  ITable CurrentTable = CurrentStructure[Table.Name];
232  Commands.Add((CurrentTable == null) ? GetTableCommand(Table) : GetAlterTableCommand(Table, CurrentTable));
233  }
234  foreach (Table Table in DesiredStructure.Tables)
235  {
236  ITable CurrentTable = CurrentStructure[Table.Name];
237  Commands.Add((CurrentTable == null) ? GetForeignKeyCommand(Table) : GetForeignKeyCommand(Table, CurrentTable));
238  Commands.Add((CurrentTable == null) ? GetTriggerCommand(Table) : GetAlterTriggerCommand(Table, CurrentTable));
239  }
240  foreach (Function Function in DesiredStructure.Functions)
241  {
242  var CurrentFunction = (Function)CurrentStructure.Functions.FirstOrDefault(x => x.Name == Function.Name);
243  Commands.Add(CurrentFunction != null ? GetAlterFunctionCommand(Function, CurrentFunction) : GetFunctionCommand(Function));
244  }
245  foreach (View View in DesiredStructure.Views)
246  {
247  var CurrentView = (View)CurrentStructure.Views.FirstOrDefault(x => x.Name == View.Name);
248  Commands.Add(CurrentView != null ? GetAlterViewCommand(View, CurrentView) : GetViewCommand(View));
249  }
250  foreach (StoredProcedure StoredProcedure in DesiredStructure.StoredProcedures)
251  {
252  var CurrentStoredProcedure = (StoredProcedure)CurrentStructure.StoredProcedures.FirstOrDefault(x => x.Name == StoredProcedure.Name);
253  Commands.Add(CurrentStoredProcedure != null ? GetAlterStoredProcedure(StoredProcedure, CurrentStoredProcedure) : GetStoredProcedure(StoredProcedure));
254  }
255  return Commands;
256  }

290  private static IEnumerable<string> GetAlterTableCommand(Table Table, ITable CurrentTable)

293  Contract.Requires<ArgumentNullException>(Table.Columns != null, "Table.Columns");
294  var ReturnValue = new List<string>();

296  {
297  IColumn CurrentColumn = CurrentTable[Column.Name];
298  string Command = "";
299  if (CurrentColumn == null)
300  {
301  Command = string.Format(CultureInfo.CurrentCulture,
302  "EXEC dbo.sp_executesql @statement = N'ALTER TABLE {0} ADD {1} {2}",
303  Table.Name,
304  Column.Name,
305  Column.DataType.To(SqlDbType.Int).ToString());
306  if (Column.DataType == SqlDbType.VarChar.To(DbType.Int32)
307  || Column.DataType == SqlDbType.NVarChar.To(DbType.Int32)
308  || Column.DataType == SqlDbType.Binary.To(DbType.Int32))
309  {
310  Command += (Column.Length < 0 || Column.Length >= 4000) ?
311  "(MAX)" :
312  "(" + Column.Length.ToString(CultureInfo.InvariantCulture) + ")";
313  }
314  else if (Column.DataType == SqlDbType.Decimal.To(DbType.Int32))
315  {
316  var Precision = (Column.Length * 2).Clamp(38, 18);
317  Command += "(" + Precision.ToString(CultureInfo.InvariantCulture) + "," + Column.Length.Clamp(38, 0).ToString(CultureInfo.InvariantCulture) + ")";
318  }
319  Command += "'";
320  ReturnValue.Add(Command);
321  foreach (IColumn ForeignKey in Column.ForeignKey)
322  {
323  Command = string.Format(CultureInfo.CurrentCulture,
324  "EXEC dbo.sp_executesql @statement = N'ALTER TABLE {0} ADD FOREIGN KEY ({1}) REFERENCES {2}({3}){4}{5}{6}'",
325  Table.Name,
326  Column.Name,
327  ForeignKey.ParentTable.Name,
328  ForeignKey.Name,
329  Column.OnDeleteCascade ? " ON DELETE CASCADE" : "",
330  Column.OnUpdateCascade ? " ON UPDATE CASCADE" : "",
331  Column.OnDeleteSetNull ? " ON DELETE SET NULL" : "");
332  ReturnValue.Add(Command);
333  }
334  }
335  else if (CurrentColumn.DataType != Column.DataType
336  || (CurrentColumn.DataType == Column.DataType
337  && CurrentColumn.DataType == SqlDbType.NVarChar.To(DbType.Int32)
338  && CurrentColumn.Length != Column.Length
339  && CurrentColumn.Length.Between(0, 4000)
340  && Column.Length.Between(0, 4000)))
341  {
342  Command = string.Format(CultureInfo.CurrentCulture,
343  "EXEC dbo.sp_executesql @statement = N'ALTER TABLE {0} ALTER COLUMN {1} {2}",
344  Table.Name,
345  Column.Name,
346  Column.DataType.To(SqlDbType.Int).ToString());
347  if (Column.DataType == SqlDbType.VarChar.To(DbType.Int32)
348  || Column.DataType == SqlDbType.NVarChar.To(DbType.Int32)
349  || Column.DataType == SqlDbType.Binary.To(DbType.Int32))
350  {
351  Command += (Column.Length < 0 || Column.Length >= 4000) ?
352  "(MAX)" :
353  "(" + Column.Length.ToString(CultureInfo.InvariantCulture) + ")";
354  }
355  else if (Column.DataType == SqlDbType.Decimal.To(DbType.Int32))
356  {
357  var Precision = (Column.Length * 2).Clamp(38, 18);
358  Command += "(" + Precision.ToString(CultureInfo.InvariantCulture) + "," + Column.Length.Clamp(38, 0).ToString(CultureInfo.InvariantCulture) + ")";
359  }

495  string Splitter = "";
496  foreach (IColumn Column in Table.Columns)
497  {
498  Builder.Append(Splitter).Append(Column.Name).Append(" ").Append(Column.DataType.To(SqlDbType.Int).ToString());
499  if (Column.DataType == SqlDbType.VarChar.To(DbType.Int32)
500  || Column.DataType == SqlDbType.NVarChar.To(DbType.Int32)
501  || Column.DataType == SqlDbType.Binary.To(DbType.Int32))
502  {
503  Builder.Append((Column.Length < 0 || Column.Length >= 4000) ?
504  "(MAX)" :
505  "(" + Column.Length.ToString(CultureInfo.InvariantCulture) + ")");
506  }
507  else if (Column.DataType == SqlDbType.Decimal.To(DbType.Int32))
508  {
509  var Precision = (Column.Length * 2).Clamp(38, 18);
510  Builder.Append("(").Append(Precision).Append(",").Append(Column.Length.Clamp(38, 0)).Append(")");
511  }
512  if (!Column.Nullable)
513  {
514  Builder.Append(" NOT NULL");
515  }
516  if (Column.Unique)
517  {
518  Builder.Append(" UNIQUE");
519  }
520  if (Column.PrimaryKey)
521  {
522  Builder.Append(" PRIMARY KEY");
523  }
524  if (!string.IsNullOrEmpty(Column.Default))
525  {
526  Builder.Append(" DEFAULT ").Append(Column.Default.Replace("(", "").Replace(")", "").Replace("'", "''"));
527  }
528  if (Column.AutoIncrement)
529  {
530  Builder.Append(" IDENTITY");
531  }
532  Splitter = ",";
533  }
534  Builder.Append(")'");
535  ReturnValue.Add(Builder.ToString());
536  int Counter = 0;
537  foreach (IColumn Column in Table.Columns)
538  {
539  if (Column.Index && Column.Unique)
540  {
541  ReturnValue.Add(string.Format(CultureInfo.CurrentCulture,
542  "EXEC dbo.sp_executesql @statement = N'CREATE UNIQUE INDEX Index_{0}{1} ON {2}({3})'",
543  Column.Name,
544  Counter.ToString(CultureInfo.InvariantCulture),
545  Column.ParentTable.Name,
546  Column.Name));
547  }
548  else if (Column.Index)
549  {
550  ReturnValue.Add(string.Format(CultureInfo.CurrentCulture,
551  "EXEC dbo.sp_executesql @statement = N'CREATE INDEX Index_{0}{1} ON {2}({3})'",
552  Column.Name,
553  Counter.ToString(CultureInfo.InvariantCulture),
554  Column.ParentTable.Name,
555  Column.Name));
