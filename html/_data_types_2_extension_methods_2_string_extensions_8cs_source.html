202  public static bool Is(this string Value, StringCompare ComparisonType)
203  {
204  if (ComparisonType == StringCompare.CreditCard)
205  {
206  long CheckSum = 0;
207  Value = Value.Replace("-", "").Reverse();
208  for (int x = 0; x < Value.Length; ++x)
209  {
210  if (!Value[x].Is(CharIs.Digit))
211  return false;
212  int TempValue = (Value[x] - '0') * (x % 2 == 1 ? 2 : 1);
213  while (TempValue > 0)
214  {
215  CheckSum += TempValue % 10;
216  TempValue /= 10;
217  }
218  }
219  return (CheckSum % 10) == 0;
220  }
221  if (ComparisonType == StringCompare.Unicode)
222  {
223  return string.IsNullOrEmpty(Value) ? true : Regex.Replace(Value, @"[^\u0000-\u007F]", "") != Value;
224  }
225  return Value.Is("", StringCompare.Anagram);
226  }

293  public static int LevenshteinDistance(this string Value1, string Value2)
294  {
295  Contract.Requires<ArgumentNullException>(Value1 != null, "Value1");
296  Contract.Requires<ArgumentNullException>(Value2 != null, "Value2");
297  int[,] Matrix = new int[Value1.Length + 1, Value2.Length + 1];
298  for (int x = 0; x <= Value1.Length; ++x)
299  Matrix[x, 0] = x;
300  for (int x = 0; x <= Value2.Length; ++x)
301  Matrix[0, x] = x;
302 
303  for (int x = 1; x <= Value1.Length; ++x)
304  {
305  for (int y = 1; y <= Value2.Length; ++y)
306  {
307  int Cost = Value1[x - 1] == Value2[y - 1] ? 0 : 1;
308  Matrix[x, y] = new int[] { Matrix[x - 1, y] + 1, Matrix[x, y - 1] + 1, Matrix[x - 1, y - 1] + Cost }.Min();
309  if (x > 1 && y > 1 && Value1[x - 1] == Value2[y - 2] && Value1[x - 2] == Value2[y - 1])
310  Matrix[x, y] = new int[] { Matrix[x, y], Matrix[x - 2, y - 2] + Cost }.Min();
311  }
312  }
313 
314  return Matrix[Value1.Length, Value2.Length];
315  }

566  public static string ToString(this string Input, StringCase Case)
567  {
568  if (string.IsNullOrEmpty(Input))
569  return "";
570  if (Case == StringCase.FirstCharacterUpperCase)
571  {
572  char[] InputChars = Input.ToCharArray();
573  for (int x = 0; x < InputChars.Length; ++x)
574  {
575  if (InputChars[x] != ' ' && InputChars[x] != '\t')
576  {
577  InputChars[x] = char.ToUpper(InputChars[x], CultureInfo.InvariantCulture);
578  break;
579  }
580  }
581  return new string(InputChars);
582  }
583  else if (Case == StringCase.SentenceCapitalize)
584  {
585  string[] Seperator = { ".", "?", "!" };
586  string[] InputStrings = Input.Split(Seperator, StringSplitOptions.None);
587  for (int x = 0; x < InputStrings.Length; ++x)
588  {
589  if (!string.IsNullOrEmpty(InputStrings[x]))
590  {
591  Regex TempRegex = new Regex(InputStrings[x]);
592  InputStrings[x] = InputStrings[x].ToString(StringCase.FirstCharacterUpperCase);
593  Input = TempRegex.Replace(Input, InputStrings[x]);
594  }
595  }
596  return Input;
597  }
598  else if (Case == StringCase.TitleCase)
599  {
600  string[] Seperator = { " ", ".", "\t", System.Environment.NewLine, "!", "?" };
601  string[] InputStrings = Input.Split(Seperator, StringSplitOptions.None);
602  for (int x = 0; x < InputStrings.Length; ++x)
603  {
604  if (!string.IsNullOrEmpty(InputStrings[x])
605  && InputStrings[x].Length > 3)
606  {
607  Regex TempRegex = new Regex(InputStrings[x].Replace(")", @"\)").Replace("(", @"\(").Replace("*", @"\*"));
608  InputStrings[x] = InputStrings[x].ToString(StringCase.FirstCharacterUpperCase);
609  Input = TempRegex.Replace(Input, InputStrings[x]);
610  }
611  }
612  return Input;
613  }
614  return Input;
615  }
