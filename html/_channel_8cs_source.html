63  XPathNavigator Node = Element.SelectSingleNode("./title", NamespaceManager);

326  public override string ToString()
327  {
328  StringBuilder ChannelString = new StringBuilder();
329  ChannelString.Append("<channel>");
330  ChannelString.Append("<title>").Append(Utils.StripIllegalCharacters(Title)).Append("</title>\r\n");
331  ChannelString.Append("<link>").Append(Link).Append("</link>\r\n");
332  ChannelString.Append("<atom:link xmlns:atom=\"http://www.w3.org/2005/Atom\" rel=\"self\" href=\"").Append(Link).Append("\" type=\"application/rss+xml\" />");
333 
334  ChannelString.Append("<description><![CDATA[").Append(Utils.StripIllegalCharacters(Description)).Append("]]></description>\r\n");
335  ChannelString.Append("<language>").Append(Language).Append("</language>\r\n");
336  ChannelString.Append("<copyright>").Append(Copyright).Append("</copyright>\r\n");
337  ChannelString.Append("<webMaster>").Append(WebMaster).Append("</webMaster>\r\n");
338  ChannelString.Append("<pubDate>").Append(PubDate.ToString("Ddd, dd MMM yyyy HH':'mm':'ss", CultureInfo.InvariantCulture)).Append("</pubDate>\r\n");
339  ChannelString.Append("<itunes:explicit>").Append((Explicit ? "yes" : "no")).Append("</itunes:explicit>");
340  ChannelString.Append("<itunes:subtitle>").Append(Utils.StripIllegalCharacters(Title)).Append("</itunes:subtitle>");
341  ChannelString.Append("<itunes:summary><![CDATA[").Append(Utils.StripIllegalCharacters(Description)).Append("]]></itunes:summary>");
342 
343  foreach (string Category in Categories)
344  {
345  ChannelString.Append("<category>").Append(Category).Append("</category>\r\n");
346  ChannelString.Append("<itunes:category text=\"").Append(Category).Append("\" />\r\n");
347  }
348  ChannelString.Append("<docs>").Append(Docs).Append("</docs>\r\n");
349  ChannelString.Append("<ttl>").Append(TTL.ToString(CultureInfo.InvariantCulture)).Append("</ttl>\r\n");
350  if (!string.IsNullOrEmpty(ImageUrl))
351  {
352  ChannelString.Append("<image><url>").Append(ImageUrl).Append("</url>\r\n<title>").Append(Title).Append("</title>\r\n<link>").Append(Link).Append("</link>\r\n</image>\r\n");
353  }
354  foreach (Item CurrentItem in Items)
355  {
356  ChannelString.Append(CurrentItem.ToString());
357  }
358  ChannelString.Append("</channel>\r\n");
359  return ChannelString.ToString();
360  }
