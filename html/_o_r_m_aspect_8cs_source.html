118  public string SetupEndMethod(MethodInfo Method, Type BaseType, string ReturnValueName)
119  {
120  StringBuilder Builder = new StringBuilder();
121  if (Mapper[BaseType] != null && Method.Name.StartsWith("get_", StringComparison.Ordinal))
122  {
123  foreach (IMapping Mapping in Mapper[BaseType])
124  {
125  IProperty Property = Mapping.Properties.FirstOrDefault(x => x.Name == Method.Name.Replace("get_", ""));
126  if (Property != null)
127  {
128  if (Property is IManyToOne || Property is IMap)
129  Builder.AppendLine(SetupSingleProperty(ReturnValueName, Property));
130  else if (Property is IIEnumerableManyToOne || Property is IManyToMany
131  || Property is IIListManyToMany || Property is IIListManyToOne
132  || Property is ICollectionManyToMany || Property is ICollectionManyToOne)
133  Builder.AppendLine(SetupIEnumerableProperty(ReturnValueName, Property));
134  else if (Property is IListManyToMany || Property is IListManyToOne)
135  Builder.AppendLine(SetupListProperty(ReturnValueName, Property));
136  return Builder.ToString();
137  }
138  }
139  }
140  return Builder.ToString();
141  }
