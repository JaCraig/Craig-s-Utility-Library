302  public SwiftBitmap ApplyConvolutionFilter(int[][] filter, bool absolute = false, int offset = 0)
303  {
304  Contract.Requires<NullReferenceException>(InternalBitmap != null);
305  using (SwiftBitmap Result = new SwiftBitmap(Width, Height))
306  {
307  Lock();
308  Result.Lock();
309  Parallel.For(0, Width, x =>
310  {
311  for (int y = 0; y < Height; ++y)
312  {
313  int RValue = 0;
314  int GValue = 0;
315  int BValue = 0;
316  int Weight = 0;
317  int XCurrent = -filter[0].Length / 2;
318  for (int x2 = 0; x2 < filter[0].Length; ++x2)
319  {
320  if (XCurrent + x < Width && XCurrent + x >= 0)
321  {
322  int YCurrent = -filter.Length / 2;
323  for (int y2 = 0; y2 < filter.Length; ++y2)
324  {
325  if (YCurrent + y < Height && YCurrent + y >= 0)
326  {
327  var Pixel = GetPixel(XCurrent + x, YCurrent + y);
328  RValue += filter[x2][y2] * Pixel.R;
329  GValue += filter[x2][y2] * Pixel.G;
330  BValue += filter[x2][y2] * Pixel.B;
331  Weight += filter[x2][y2];
332  }
333  ++YCurrent;
334  }
335  }
336  ++XCurrent;
337  }
338  var MeanPixel = GetPixel(x, y);
339  if (Weight == 0)
340  Weight = 1;
341  if (Weight > 0)
342  {
343  if (absolute)
344  {
345  RValue = System.Math.Abs(RValue);
346  GValue = System.Math.Abs(GValue);
347  BValue = System.Math.Abs(BValue);
348  }
349  RValue = (RValue / Weight) + offset;
350  RValue = RValue.Clamp(255, 0);
351  GValue = (GValue / Weight) + offset;
352  GValue = GValue.Clamp(255, 0);
353  BValue = (BValue / Weight) + offset;
354  BValue = BValue.Clamp(255, 0);
355  MeanPixel = Color.FromArgb(RValue, GValue, BValue);
356  }
357  Result.SetPixel(x, y, MeanPixel);
358  }
359  });
360  Unlock();
361  Result.Unlock();
362  return Copy(Result);
363  }
