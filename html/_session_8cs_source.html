204  public List<DataType> LoadProperties<ObjectType, DataType>(ObjectType Object, string PropertyName)
205  where ObjectType : class,new()
206  where DataType : class,new()
207  {
208  System.Collections.Generic.List<Dynamo> ReturnValue = new System.Collections.Generic.List<Dynamo>();
209  foreach (ISourceInfo Source in SourceProvider.Where(x => x.Readable).OrderBy(x => x.Order))
210  {
211  IMapping Mapping = MapperProvider[typeof(ObjectType), Source];
212  if (Mapping != null)
213  {
214  IProperty Property = Mapping.Properties.FirstOrDefault(x => x.Name == PropertyName);
215  if (Property != null)
216  {
217  foreach (Dynamo Item in QueryProvider.Generate<ObjectType>(Source, Mapping)
218  .LoadProperty<DataType>(Object, Property)
219  .Execute()[0])
220  {
221  IProperty IDProperty = Property.ForeignMapping.IDProperties.FirstOrDefault();
222  CopyOrAdd(ReturnValue, IDProperty, Item);
223  }
224  }
225  }
226  }
227  if (ReturnValue.Count == 0)
228  return new List<DataType>();
229  foreach (ISourceInfo Source in SourceProvider.Where(x => x.Readable).OrderBy(x => x.Order))
230  {
231  IMapping ObjectMapping = MapperProvider[typeof(ObjectType), Source];
232  IMapping Mapping = MapperProvider[typeof(DataType), Source];
233  if (Mapping != null)
234  {
235  IProperty ObjectProperty = ObjectMapping == null ? null : ObjectMapping.Properties.FirstOrDefault(x => x.Name == PropertyName);
236  if (ObjectProperty == null)
237  {
238  IProperty IDProperty = Mapping.IDProperties.FirstOrDefault();
239  IParameter Parameter = null;
240  int Counter = 0;
241  foreach (Dynamo Item in ReturnValue)
242  {
243  if (IDProperty.GetParameter(Item) != null)
244  {
245  if (Parameter == null)
246  Parameter = new EqualParameter<object>(IDProperty.GetParameter(Item), Counter.ToString(CultureInfo.InvariantCulture), IDProperty.FieldName, Source.ParameterPrefix);
247  else
248  Parameter = new OrParameter(Parameter, new EqualParameter<object>(IDProperty.GetParameter(Item), Counter.ToString(CultureInfo.InvariantCulture), IDProperty.FieldName, Source.ParameterPrefix));
249  ++Counter;
250  }
251  }
252  if (Parameter != null)
253  {
254  foreach (Dynamo Item in QueryProvider.Generate<DataType>(Source, Mapping).All(Parameter).Execute()[0])
255  {
256  CopyOrAdd(ReturnValue, IDProperty, Item);
257  }
258  }
259  }
260  }
261  }
262  return ConvertValues<DataType>(ReturnValue).ToList();
263  }

411  private static void CopyOrAdd(List<Dynamo> ReturnValue, IProperty IDProperty, Dynamo Item)
412  {
413  Contract.Requires<ArgumentNullException>(IDProperty != null);
414  if (Item == null)
415  return;
416  if (ReturnValue == null)
417  ReturnValue = new List<Dynamo>();
418  object IDValue = IDProperty.GetValue(Item);
419  Dynamo Value = ReturnValue.FirstOrDefault(x => IDProperty.GetValue(x).Equals(IDValue));
420  if (Value == null)
421  ReturnValue.Add(Item);
422  else
423  Item.CopyTo(Value);
424  }
