109  private void FindClosestPoint(int x, int y)
110  {
111  Contract.Requires<NullReferenceException>(Points != null, "Points");
112  float MaxDistance = float.MaxValue;
113  int Index = -1;
114  for (int z = 0; z < Points.Count; ++z)
115  {
116  float Distance = (float)System.Math.Sqrt(((Points[z].X - x) * (Points[z].X - x)) + ((Points[z].Y - y) * (Points[z].Y - y)));
117  if (Distance < MaxDistance)
118  {
119  MaxDistance = Distance;
120  Index = z;
121  }
122  }
123  ClosestPoint[x, y] = Index;
124  Distances[x, y] = MaxDistance;
125  if (MaxDistance < this.MinDistance)
126  this.MinDistance = MaxDistance;
127  if (MaxDistance > this.MaxDistance)
128  this.MaxDistance = MaxDistance;
129  }
