108  public AssetType DetermineType(string Path)
109  {
110  AssetType Type = Translators.FirstOrDefault(x => Path.EndsWith(x.FileTypeAccepts, StringComparison.Ordinal))
111  .Chain(x => x.TranslatesTo, AssetType.Unknown);
112  if (Type == AssetType.Unknown && Path.EndsWith("css", StringComparison.Ordinal))
113  return AssetType.CSS;
114  else if (Type == AssetType.Unknown && Path.EndsWith("js", StringComparison.Ordinal))
115  return AssetType.Javascript;
116  return Type;
117  }

124  public void Process(IList<IAsset> Assets, BundleResponse Response)
125  {
126  Contract.Requires<ArgumentNullException>(Response != null, "Response");
127  if (Assets == null || Assets.Count == 0)
128  return;
129  foreach (IFilter Filter in Filters.Where(x => x.TimeToRun == RunTime.PreTranslate))
130  {
131  Assets = Filter.Filter(Assets);
132  }
133  foreach (ITranslator Translator in Translators)
134  {
135  Assets = Translator.Translate(Assets);
136  }
137  foreach (RunTime Item in RunOrder)
138  {
139  foreach (IFilter Filter in Filters.Where(x => x.TimeToRun == Item))
140  {
141  Assets = Filter.Filter(Assets);
142  }
143  }
144  string Content = Assets.OrderBy(x => x.Path).ToString(x => x.ToString(), "");
145  foreach (IContentFilter Filter in ContentFilters)
146  {
147  Content = Filter.Filter(Content);
148  }
149  var Files = new List<BundleFile>();
150  foreach (IAsset Asset in Assets)
151  {
152  if (Asset.Path.StartsWith("~", StringComparison.Ordinal) || Asset.Path.StartsWith("/", StringComparison.Ordinal))
153  {
154  Files.Add(new BundleFile(Asset.Path, new VirtualFileHack(Asset.Path)));
155  Files.Add(Asset.Included.Where(x => x.Path.StartsWith("~", StringComparison.Ordinal) || x.Path.StartsWith("/", StringComparison.Ordinal)).Select(x => new BundleFile(x.Path, new VirtualFileHack(x.Path))));
156  }
157  }
158  Response.Content = Content;
159  Response.Files = Files.OrderBy(x => x.VirtualFile.VirtualPath);
160  Response.ContentType = Assets.First().Type == AssetType.CSS ? "text/css" : "text/javascript";
161  Response.Cacheability = HttpCacheability.ServerAndPrivate;
162  }

181  private void CreateBundles(DirectoryInfo Directory)
182  {
183  if (Directory == null || !Directory.Exists || string.IsNullOrEmpty(Directory.FullName))
184  return;
185  string BundleDirectory = Directory.FullName.Replace(new DirectoryInfo("~/").FullName, "~/").Replace("\\", "/");
186  var Bundle = new StyleBundle(BundleDirectory + "/bundle/css");
187  Bundle.Transforms.Clear();
188  Bundle.Transforms.Add(new Transformer(this));
189  if (Directory.Exists)
190  {
191  foreach (string Value in FileTypes[AssetType.CSS])
192  {
193  Bundle.IncludeDirectory(BundleDirectory, "*." + Value, true);
194  }
195  }
196  var Bundle2 = new ScriptBundle(BundleDirectory + "/bundle/js");
197  Bundle2.Transforms.Clear();
198  Bundle2.Transforms.Add(new Transformer(this));
199  if (Directory.Exists)

202  {
203  Bundle2.IncludeDirectory(BundleDirectory, "*." + Value, true);
204  }
205  }
206  BundleTable.Bundles.Add(Bundle);
207  BundleTable.Bundles.Add(Bundle2);
