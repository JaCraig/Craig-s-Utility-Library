108  protected virtual ActionResult Serialize<T>(T Object, string ContentType = "")
109  {
110  Utilities.IO.Serializers.Manager Manager = Utilities.IoC.Manager.Bootstrapper == null ? null : Utilities.IoC.Manager.Bootstrapper.Resolve<Utilities.IO.Serializers.Manager>();
111  ContentResult Result = new ContentResult();
112  if (Manager == null)
113  return Result;
114  if (string.IsNullOrEmpty(ContentType))
115  {
116  if (Request.AcceptTypes != null)
117  ContentType = Request.AcceptTypes.Length > 0 ? Request.AcceptTypes.FirstOrDefault(x => Manager.CanSerialize(x)) : "";
118  if (string.IsNullOrEmpty(ContentType) && Request.Path.Contains('.'))
119  ContentType = Manager.FileTypeToContentType(Request.Path.ToUpperInvariant().Right((Request.Path.Length - Request.Path.LastIndexOf('.'))));
120  if (string.IsNullOrEmpty(ContentType))
121  ContentType = "application/json";
122  }
123  Result.Content = Object.Serialize<string, T>(ContentType);
124  Result.ContentType = ContentType;
125  return Result;
126  }
