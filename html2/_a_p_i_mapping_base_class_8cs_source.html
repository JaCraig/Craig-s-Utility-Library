142  public IEnumerable<Dynamo> All(MappingHolder Mappings, params string[] EmbeddedProperties)
143  {
144  IEnumerable<ClassType> Objects = AllFunc();
145  if (Objects == null)
146  Objects = new List<ClassType>();
147  var ReturnValue = new List<Dynamo>();
148  foreach (ClassType Object in Objects)
149  {
150  if (CanGetFunc(Object))
151  {
152  var TempItem = new Dynamo(Object);
153  Dynamo ReturnItem = TempItem.SubSet(Properties.Where(x => x is IReference || x is IID)
154  .Select(x => x.Name)
155  .ToArray());
156  string AbsoluteUri = HttpContext.Current != null ? HttpContext.Current.Request.Url.AbsoluteUri.Left(HttpContext.Current.Request.Url.AbsoluteUri.IndexOf('?')) : "";
157  AbsoluteUri = AbsoluteUri.Check("");
158  if (!AbsoluteUri.EndsWith("/", StringComparison.Ordinal))
159  AbsoluteUri += "/";
160  AbsoluteUri += Properties.FirstOrDefault(x => x is IID).GetValue(Mappings, Object).ToString() + "/";
161  foreach (IAPIProperty Property in Properties.Where(x => x is IMap))
162  {
163  ReturnItem[Property.Name] = EmbeddedProperties.Contains(Property.Name) ?
164  (object)Property.GetValue(Mappings, TempItem) :
165  (object)(AbsoluteUri + Property.Name);
166  }
167  ReturnValue.Add(ReturnItem);
168  }
169  }
170  return ReturnValue;
171  }

413  public bool Save(Dynamo Object)
414  {
415  if (Object == null)
416  return false;
417  IAPIProperty IDProperty = Properties.FirstOrDefault(x => x is IID);
418  if (IDProperty == null)
419  return false;
420  object IDValue = Object[IDProperty.Name];
421  ClassType Item = IDValue == null ? new ClassType() : AnyFunc(IDValue.ToString()).Check(new ClassType());
422  if (!CanSaveFunc(Item))
423  return false;
424  Dynamo SubSet = Object.SubSet(Properties.Where(x => x is IReference)
425  .Select(x => x.Name)
426  .ToArray());
427  if (SubSet != null)
428  SubSet.CopyTo(Item);
429  return SaveFunc(Item);
430  }
