156  public bool InstallPlugin(string ID)
157  {
158  Contract.Requires<ArgumentNullException>(!string.IsNullOrEmpty(ID), "ID");
159  string User = HttpContext.Current.Chain(x => x.User).Chain(x => x.Identity).Chain(x => x.Name, "");
160  Utilities.IO.Log.Get().LogMessage("Plugin {0} is being installed by {1}", MessageType.Debug, ID, User);
161  Plugin TempPlugin = PluginList.Get(ID);
162  if (TempPlugin != null)
163  UninstallPlugin(ID);
164  foreach (IPackageRepository Repo in PackageRepositories)
165  {
166  IPackage Package = Repo.FindPackage(ID);
167  if (Package != null)
168  {
169  new DirectoryInfo(AppDomain.CurrentDomain.BaseDirectory + "/App_Data/packages/" + Package.Id + "." + Package.Version.ToString() + "/lib").Create();
170  new DirectoryInfo(AppDomain.CurrentDomain.BaseDirectory + "/App_Data/packages/" + Package.Id + "." + Package.Version.ToString() + "/content").Create();
171  new DirectoryInfo(AppDomain.CurrentDomain.BaseDirectory + "/App_Data/packages/" + Package.Id + "." + Package.Version.ToString() + "/tools").Create();
172  PackageManager Manager = new PackageManager(Repo,
173  new DefaultPackagePathResolver(Repo.Source),
174  new PhysicalFileSystem(new DirectoryInfo(HttpContext.Current != null ?
175  HttpContext.Current.Server.MapPath("~/App_Data/packages") :
176  "./App_Data/packages").FullName));
177  Manager.InstallPackage(Package, false, true);
178  PluginList.Add(new Plugin(Package));
179  Package.DependencySets.ForEach(x => x.Dependencies.ForEach(y => InstallPlugin(y.Id)));
180  PluginList.Save();
181  break;
182  }
183  }
184  Utilities.IO.Log.Get().LogMessage("Plugin {0} has been installed by {1}", MessageType.Debug, ID, User);
185  return true;
186  }

213  public bool UpdatePlugin(string ID)
214  {
215  Contract.Requires<ArgumentNullException>(!string.IsNullOrEmpty(ID), "ID");
216  Plugin TempPlugin = PluginList.Get(ID);
217  if (TempPlugin == null)
218  return true;
219  string User = HttpContext.Current.Chain(x => x.User).Chain(x => x.Identity).Chain(x => x.Name, "");
220  bool Result = false;
221  Utilities.IO.Log.Get().LogMessage("Plugin {0} is being updated by {1}", MessageType.Debug, ID, User);
222  foreach (IPackageRepository Repo in PackageRepositories)
223  {
224  IPackage Package = Repo.FindPackage(ID);
225  if (Package != null)
226  {
227  TempPlugin = PluginList.Get(ID);
228  if (TempPlugin != null)
229  {
230  TempPlugin.OnlineVersion = Package.Version.ToString();
231  TempPlugin.Save();
232  if (TempPlugin.UpdateAvailable)
233  Result = UninstallPlugin(ID);
234  if (Result)
235  Result = InstallPlugin(ID);
236  }
237  break;
238  }
